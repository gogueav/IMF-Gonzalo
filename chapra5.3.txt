%EJERCICIO 03 PROBLEMAS PROPUESTOS
% a)Gráficamente 
%   ezplot('-25182*x-90*x^2+44*x^3-8*x^4+0.7*x^5',[-5,5,-5,5])
% b) Usando el método de bisección para localizar la raíz más
% grande con es = 10%. Utilice como valores iniciales xl = 0.5
% y xu = 1.0.
f='-25182*x-90*x^2+44*x^3-8*x^4+0.7*x^5';
f=inline(f);
%Introduciendo valores
xl=input('Introduzca el límite inferior: ');
xu=input('Introduzca el límite superior: ');
es=input('Introduzca error estimado: ');
imax=input('¿Cantidad máxima de iteraciones?: ');
i=1;
ea=1;
xr=0;
R=1;
disp(['i      xl        xu         xr         ea       '])
while (i<=imax)&&(ea>es)
    xrold=xr;
    xr=(xl+xu)/2;
    ea=abs((xu-xl)/(xu+xl));
    EA=abs((xr-xrold)/xr)*100;
    fprintf('\n%2d\t %2.6f\t%2.6f\t%2.6f\t%2.6f\t%2.6f',i,xl,xu,xr,ea,EA)
    i=i+1;
   %Et=abs(((x2-xr)/(x2))*100);
    R=f(xl)*f(xr);
    if R>0
        xl=xr;
    elseif R<0
        xu=xr;
    else
        ea=0;
    end
end
