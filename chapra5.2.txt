clear,clc
%EJERCICIO 02 PROBLEMAS PROPUESTOS
% a) ezplot('5*x^3-5*x^2+6*x-2',[-5,5,-5,5])
% b) Utilizando el método de bisección para localizar la raíz más
% pequeña. Use los valores iniciales xl = 0 y xu = 1 iterando
% hasta que el error estimado ea se encuentre debajo de es =10%.
f='5*x^3-5*x^2+6*x-2';
f=inline(f);
%Introduciendo valores
xl=input('Introduzca el límite inferior: ');
xu=input('Introduzca el límite superior: ');
es=input('Introduzca error estimado: ');
imax=input('¿Cantidad máxima de iteraciones?: ');
ea=1;
xr=0;
i=1;
disp(['i       xl        xu         xr         ea         EA '])
while (i<=imax)&&(ea>es)
    xrold=xr;
    xr=(xl+xu)/2;
    ea=abs((xu-xl)/(xu+xl));%chapra
    EA=abs((xr-xrold)/xr)*100;%método profe
    R=f(xl)*f(xr);
    fprintf('\n%2d\t %2.6f\t%2.6f\t%2.6f\t%2.6f\t%2.5f',i,xl,xu,xr,ea,EA)
%     disp([xl,xu,xr,ea,EA])
    i=i+1;
   %Et=abs(((x2-xr)/(x2))*100);
    if R>0
        xl=xr;
    elseif R<0
        xu=xr;
    else
        ea=0;
    end
end
